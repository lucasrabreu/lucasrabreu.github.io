class MyLDA:
    def fit(self, df, target_col):
        self.target_col = target_col
        X = df.drop(target_col, axis=1)
        y = df[target_col]
        
        # Sb and sw declaration
        n_features = X.shape[1]
        sb = np.zeros((n_features, n_features))
        sw = np.zeros((n_features, n_features))

        # Grand mean vector (X bar)
        w, h = X.shape
        centroide = X.mean()

        # for each class
        for ix, gp in df.groupby(target_col):
            X_i = gp.drop(target_col, axis=1)
            
            # Xi bar
            class_mean = X_i.mean()

            # between-class scatter matrix Sb be defined as:
            sb_i = (class_mean - centroide).values.reshape(-1, 1)
            sb += len(gp) * (sb_i @ sb_i.T)

            # within-class scatter matrix Sw be defined as:
            sw_i = (X_i - class_mean).values
            sw += sw_i.T @ sw_i
            
        # calc eigenvalues and eigenvectors
        self.w, self.v = np.linalg.eig(np.linalg.inv(sw) @ sb)

        # calc best components
        self.variance_explained = self.w/self.w.sum()
        print('% best LDAs: {}'.format(self.variance_explained.round(4)*100))
        
        self.sw = sw
        self.sb = sb
        
    def transform(self, X, n_components=2):
        X = X.drop(self.target_col, axis=1, errors='ignore')

        # select best components
        self.W = self.v[:, self.w.argsort()[::-1][:n_components]]
        transformed = X @ self.W
        pcas = pd.DataFrame(transformed, 
                            columns=[f'LDA{i}' for i in range(1, n_components+1)], index=X.index)

        return pcas

Leitura de dados¶

# Leitura de dados
df = pd.read_html('http://en.wikipedia.org/wiki/Iris_flower_data_set', encoding='UTF-8', index_col=0, header=0)[0]

# Tratamento da label
df.Species = df.Species.str.split('\xa0', expand=True)[1].str.capitalize()

X = df.drop('Species', axis=1)
y = df.Species.factorize()[0]

Fit dos algoritmos¶

In [4]:
# LDA implementado
mylda = MyLDA()
mylda.fit(df, 'Species')
my_result = mylda.transform(df, 2)
my_result['Species'] = df.Species

# LDA do scikit-learn
lda = LinearDiscriminantAnalysis(solver='eigen', n_components=2)
lda.fit(X, y)
sklearn_result = pd.DataFrame(lda.transform(-X))
sklearn_result['Species'] = df.Species
